На какой машине: Локальная машина (Windows 11 Home Edition)

Описание

В этом шаге мы клонируем репозиторий проекта с GitHub на вашу локальную машину для дальнейшей работы.

Предварительные требования

  • Установленный Git.
  • Настроенный аккаунт на GitHub.
  • Репозиторий проекта создан на GitHub (см. предыдущий шаг).

Действия:

  1. Перейдите на официальный сайт Git: https://git-scm.com/download/win
  2. Скачайте установщик для Windows и запустите его.
  3. Следуйте инструкциям установщика, оставив настройки по умолчанию.
  4. После установки перезагрузите командную строку, если она была открыта.
  5. Проверьте установку Git, выполнив команду:
    6. git --version

Действия:

  1. Откройте командную строку или PowerShell.
  2. Перейдите в директорию, где вы хотите разместить проект:
    3. cd C:\Users\ВашеИмяПользователя\Projects
  3. Клонируйте репозиторий проекта:
    4. git clone https://github.com/yourusername/{{PROJECT_NAME}}.git

    Замените yourusername на ваше имя пользователя на GitHub, а {{PROJECT_NAME}} на название вашего проекта.

  4. Перейдите в директорию проекта:
    5. cd {{PROJECT_NAME}}

Чек-лист: Проверка успешного клонирования

  • Репозиторий проекта склонирован на вашу локальную машину.
  • Вы находитесь в директории проекта.
  • Файлы проекта видны в файловом менеджере или через команду dir.

Полезные ссылки

  • Документация Git
  • Руководство по началу работы с GitHub

Описание

Установим все необходимые пакеты и зависимости для запуска проекта.

Предварительные требования

  • Находитесь в директории проекта.
  • Установлены Node.js и npm.

Действия:

  1. Убедитесь, что файл package.json существует в корне проекта.
  2. Выполните команду для установки зависимостей:
    3. npm install

    Эта команда установит все пакеты, перечисленные в секциях dependencies и devDependencies файла package.json.

Объяснение файла package.json

Файл package.json содержит информацию о вашем проекте и списке зависимостей. Пример:

{
  "name": "{{PROJECT_NAME}}",
  "version": "1.0.0",
  "description": "Описание вашего проекта",
  "main": "dist/server/index.js",
  "scripts": {
    "start": "node dist/server/index.js",
    "build": "gulp build",
    "dev": "gulp"
  },
  "author": "Ваше имя",
  "license": "ISC",
  "dependencies": {
    "express": "^4.17.1",
    "mongoose": "^5.13.8"
  },
  "devDependencies": {
    "@types/express": "^4.17.13",
    "@types/mongoose": "^5.11.97",
    "gulp": "^4.0.2",
    "gulp-pug": "^4.0.1",
    "gulp-sass": "^5.0.0",
    "gulp-typescript": "^6.0.0-alpha.1",
    "sass": "^1.32.13",
    "typescript": "^4.5.4",
    "del": "^6.0.0",
    "browser-sync": "^2.26.14",
    "gulp-clean-css": "^4.3.0",
    "gulp-uglify": "^3.0.2"
  }
}

Объяснение:

  • dependencies: Пакеты, необходимые для работы приложения в продакшене.
  • devDependencies: Пакеты, необходимые для разработки (инструменты сборки, типизация и т.д.).
  • scripts: Сценарии для автоматизации задач (сборка, запуск сервера и т.д.).

Чек-лист: Проверка установки зависимостей

  • В директории проекта появилась папка node_modules.
  • Команда npm install выполнилась без ошибок.

Описание

В этом шаге мы настроим и интегрируем Pug, Sass и Gulp в наш проект для улучшения процесса разработки и автоматизации задач.

Действия:

  1. Установите Pug и плагин для Gulp:
    2. npm install --save-dev pug gulp-pug
  2. Создайте директорию для шаблонов Pug:
    3. mkdir src/client/views
  3. Создайте базовый шаблон src/client/views/index.pug:
    4. doctype html
html(lang="ru")
  head
    meta(charset="UTF-8")
    title Мой проект
    link(rel="stylesheet", href="css/main.css")
  body
    h1 Здравствуйте, мир!
    p Это мой первый шаблон Pug.
    script(src="js/app.js")

Действия:

  1. Установите Sass и плагин для Gulp:
    2. npm install --save-dev sass gulp-sass
  2. Создайте директорию для файлов Sass:
    3. mkdir src/client/styles
  3. Создайте файл src/client/styles/main.scss с базовыми стилями:
    4. $primary-color: #3498db;

body {
  font-family: Arial, sans-serif;
  background-color: #ecf0f1;
}

h1 {
  color: $primary-color;
}

Действия:

  1. Установите Gulp глобально и в проекте:
    2. npm install --global gulp-cli
npm install --save-dev gulp
  2. Установите BrowserSync для автоматического обновления браузера:
    3. npm install --save-dev browser-sync
  3. Создайте или отредактируйте файл gulpfile.js и добавьте следующие задачи: 
    4. // Импорт необходимых модулей
const gulp = require('gulp');
const pug = require('gulp-pug');
const sass = require('gulp-sass')(require('sass'));
const ts = require('gulp-typescript');
const tsProject = ts.createProject('tsconfig.json');
const browserSync = require('browser-sync').create();
const del = require('del');

// Очистка папки dist
function clean() {
  return del(['dist']);
}

// Компиляция Pug
function compilePug() {
  return gulp.src('src/client/views/**/*.pug')
    .pipe(pug())
    .pipe(gulp.dest('dist/public'))
    .pipe(browserSync.stream());
}

// Компиляция Sass
function compileSass() {
  return gulp.src('src/client/styles/**/*.scss')
    .pipe(sass().on('error', sass.logError))
    .pipe(gulp.dest('dist/public/css'))
    .pipe(browserSync.stream());
}

// Компиляция клиентского TypeScript
function compileClientTs() {
  return gulp.src('src/client/**/*.ts')
    .pipe(tsProject())
    .pipe(gulp.dest('dist/public/js'))
    .pipe(browserSync.stream());
}

// Компиляция серверного TypeScript
function compileServerTs() {
  return gulp.src('src/server/**/*.ts')
    .pipe(tsProject())
    .pipe(gulp.dest('dist'))
    .pipe(browserSync.stream());
}

// Копирование статических файлов
function copyAssets() {
  return gulp.src('public/**/*')
    .pipe(gulp.dest('dist/public'))
    .pipe(browserSync.stream());
}

// Запуск сервера BrowserSync
function serve() {
  browserSync.init({
    server: {
      baseDir: 'dist/public'
    }
  });

  gulp.watch('src/client/views/**/*.pug', compilePug);
  gulp.watch('src/client/styles/**/*.scss', compileSass);
  gulp.watch('src/client/**/*.ts', compileClientTs);
  gulp.watch('src/server/**/*.ts', compileServerTs);
  gulp.watch('public/**/*', copyAssets);
}

// Задача по умолчанию
exports.default = gulp.series(
  clean,
  gulp.parallel(compilePug, compileSass, compileClientTs, compileServerTs, copyAssets),
  serve
);

// Задача для сборки без запуска сервера
exports.build = gulp.series(
  clean,
  gulp.parallel(compilePug, compileSass, compileClientTs, compileServerTs, copyAssets)
);

  • Sass установлен и настроен в проекте.
  • Gulp настроен для автоматизации задач компиляции и запуска сервера.

Описание

В этом шаге мы настроим конфигурационные файлы проекта: tsconfig.json и gulpfile.js.

Предварительные требования

  • Установлены все зависимости проекта.

Действия:

  1. Создайте файл tsconfig.json в корне проекта, если он не был создан ранее.
  2. Добавьте следующий код:
    3. {
  "compilerOptions": {
    "target": "ES6",
    "module": "commonjs",
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules"]
}

    Объяснение:

    • "outDir": "./dist": Папка для скомпилированных файлов.
    • "rootDir": "./src": Папка с исходными файлами TypeScript.
    • "include": ["src/**/*"]: Компилировать все файлы внутри src.

Настроим Gulp для автоматической компиляции Pug, Sass и TypeScript, а также для копирования статических файлов.

Действия:

  1. Создайте файл gulpfile.js в корне проекта.
  2. Добавьте следующий код:
    3. // Импортируем необходимые модули
const { src, dest, watch, series } = require('gulp');
const pug = require('gulp-pug');
const sass = require('gulp-sass')(require('sass'));
const ts = require('gulp-typescript');
const tsProject = ts.createProject('tsconfig.json');
const del = require('del');

// Очистка папки dist перед сборкой
function cleanDist() {
  return del(['dist/**', '!dist']);
}

// Компиляция Pug в HTML
function compilePug() {
  return src('src/client/**/*.pug')
    .pipe(pug())
    .pipe(dest('dist/public'));
}

// Компиляция Sass в CSS
function compileSass() {
  return src('src/client/**/*.scss')
    .pipe(sass().on('error', sass.logError))
    .pipe(dest('dist/public/css'));
}

// Компиляция TypeScript (клиентская часть)
function compileClientTS() {
  return src('src/client/**/*.ts')
    .pipe(tsProject())
    .pipe(dest('dist/public/js'));
}

// Компиляция TypeScript (серверная часть)
function compileServerTS() {
  return src('src/server/**/*.ts')
    .pipe(tsProject())
    .pipe(dest('dist'));
}

// Копирование статических файлов
function copyAssets() {
  return src('public/**/*')
    .pipe(dest('dist/public'));
}

// Наблюдение за изменениями
function watchFiles() {
  watch('src/client/**/*.pug', compilePug);
  watch('src/client/**/*.scss', compileSass);
  watch('src/client/**/*.ts', compileClientTS);
  watch('src/server/**/*.ts', compileServerTS);
  watch('public/**/*', copyAssets);
}

// Экспорт задач
exports.default = series(
  cleanDist,
  compilePug,
  compileSass,
  compileClientTS,
  compileServerTS,
  copyAssets,
  watchFiles
);

exports.build = series(
  cleanDist,
  compilePug,
  compileSass,
  compileClientTS,
  compileServerTS,
  copyAssets
);

    Объяснение:

    • cleanDist: Очищает папку dist перед сборкой.
    • compilePug, compileSass, compileClientTS, compileServerTS: Компилируют соответствующие файлы.
    • copyAssets: Копирует статические файлы из папки public в dist/public.
    • watchFiles: Следит за изменениями в файлах и запускает соответствующие задачи.
    • exports.default: Задача по умолчанию для разработки.
    • exports.build: Задача для сборки проекта без наблюдения.

Чек-лист: Проверка конфигурации

  • Файл tsconfig.json настроен правильно.
  • Файл gulpfile.js содержит все необходимые задачи.

  • Руководство по Gulp
